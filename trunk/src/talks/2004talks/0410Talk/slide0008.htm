8 How to Use DBI

Here's a little DBI example 

    
#!/usr/bin/perl

use DBI;

use strict;
use warnings;

my $dbh = DBI->connect( "dbi:CSV:f_dir=db", "fred", "secret" )
    or die "Cannot connect to database: $DBI::errstr\n";

my $sth = $dbh->prepare( "SELECT * from People" )
    or die "Cannot prepare select statement: $DBI::errstr\n";

$sth->execute()
    or die "Cannot execute select statement: $DBI::errstr\n";

my @row;
while(@row = $sth->fetchrow_array())
{
    print "Row: ", join( ", ", @row ), "\n";
}

$dbh->disconnect;
