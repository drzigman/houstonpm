2.3 Other Anchors

    \A - always beginning of string
    \z - always end of string
    \Z - end of string or right before newline at end of string 
    

  my $string = "This is a\nmulti-line string\n";
  $string =~ /\A\w+/;    # matches 'This'
  $string =~ /\A[a-z]+/; # doesn't match
  $string =~ /\w+\Z/m;   # matches "string\n"
  $string =~ /\w+\z/m;   # doesn't match because of the "\n"
