5.1 Options

    m - multi-line, let ^ and $ match next to embedded newlines 
    
  my $str = "abcd\nabc 123\n456";
  $str = /\d+$/;  # matches 456
  $str = /^\d+/;  # doesn't match
  $str = /\d+$/m; # matches 123
  $str = /^\d+/m; # matches 456
    s - single line, let . match newline 
    
  my $str = "abcd\nefg";
  $str = /.*/;  # matches abcd
  $str = /.*/s; # matches abcd\nefg
    x - ignore whitespace and comments in expression
