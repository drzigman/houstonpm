5.3 Basic Extensions

    (?:) - non-capturing cluster 
    
   my $string = "fred, wilma, barney, and betty";

   my @chars = (/(\w+)(?:, (?:and )?)?/g);
    (?imsx-imsx) - apply options to a subset of the expression 
    
   /\d\d?:\d\d\s*(?i:am|pm)/; # match a time
