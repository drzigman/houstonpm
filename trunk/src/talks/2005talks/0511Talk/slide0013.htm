10 Inline::Java Example 

    
#!/usr/bin/perl

use Inline Java => 'DATA';

use strict;
use warnings;


my $obj = new InlineTest( 17 );
$obj->incr();
$obj->incr();
$obj->incr();

$obj->print();

__DATA__

__Java__

class InlineTest
{
   int value_;

   public InlineTest( int value )
   {
       value_ = value;
   }
   
   public void incr()
   {
       value_++;
   }
   
   public void print()
   {
       System.out.println( "This is a test: " + value_ );
   }
}
