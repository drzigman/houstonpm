7 Inline::C Example 

    
#!/usr/bin/perl

use Inline C => 'DATA';

use strict;
use warnings;

hello_world( $$ ^ time );

for my $i (0 .. 10)
{
    my $roll = roll_dice( 3 );
    print "$i:\t$roll\n";
}

__DATA__
__C__

#include <stdio.h>
#include <stdlib.h>

/*
 * Example C code
 */

void  hello_world( unsigned seed )
{
    srand( seed );
    printf( "Hello world!\n" );
}


unsigned int  roll_dice( unsigned int count )
{
    unsigned int sum = 0;
    int i =0;

    for(i = 0; i < count;++i)
    {
        /* Low order bits are often not very interesting. */
        sum += (rand() % 6) + 1;
    }

    return sum;
}
