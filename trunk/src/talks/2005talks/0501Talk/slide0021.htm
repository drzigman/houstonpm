8 Using the Class 

    
#!/usr/bin/perl

use Test::More tests => 11;

use strict;
use warnings;

use ClassDerived;

is( ClassDerived::howMany(), 0, "None instantiated" );

my $obj = new ClassDerived();
isa_ok( $obj, 'ClassDerived' );
isa_ok( $obj, 'ClassSimple' );

is( ClassDerived::howMany(), 1, "One instantiated" );

is( $obj->get(), 0, "Initial value okay" );

$obj->add( 1 );
is( $obj->get(), 1, "One addition works." );

$obj->incr();
is( $obj->get(), 2, "Increment works." );

$obj->set( 5 );
is( $obj->get(), 5, "Set works." );

{
    my $other = new ClassDerived();
    is( ClassDerived::howMany(), 2, "Object in a block." );
}
is( ClassDerived::howMany(), 1, "Count correct after block." );

undef $obj;

is( ClassDerived::howMany(), 0, "All Destoryed" );
