12.3 Simpler Substitutes 

    
  # which one removes leading and trailing whitespace.
  s/\s+(.*)\s+/$1;
    # or
  s/\s*(.*?)\s*/$1/;
    # or
  s/^\s*(.*)\s*$/$1/;
    # or
  s/^\s*(.*?\S)\s*$/$1/;
    # or
  s/^\s+//;
  s/\s+$//;
