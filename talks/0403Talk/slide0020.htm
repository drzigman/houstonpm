Example

The flexibility and power of Getopt::Declare means that we'll only get a superficial overview, and the best way to begin is by simply showing an example: 

    
    #!/usr/bin/perl -w
#    use strict;
    use Getopt::Declare;
    use vars qw/$not $re/;
    $::VERSION = 0.01;

    my $spec = <<'EOS';

        -p <pattern>		pattern [required]
                                { $re = $pattern }

        -f <INFILE>...		input filename(s)  [required]
                                { defer{process(@INFILE)} }

        -not			print out non-matches
                                { $not = 1 }
    EOS

    my $opts = Getopt::Declare->new($spec);
    print Getopt::Declare::version();

    sub process {
        @ARGV = @_;
        while(<>){
            if($::not){
                print unless /$::re/;
            } else {
                print if /$::re/;
            }
        }
    }
    __END__
