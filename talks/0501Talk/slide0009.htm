5 A Simple Perl Class 

    
# The really Simple Class

package ClassSimple;

use strict;
use warnings;

sub  new
{
    bless { counter => 0 };
}

sub  add
{
    my $self = shift;
    $self->{counter} += shift;
}

sub  get
{
    my $self = shift;
    $self->{counter};
}

1;
