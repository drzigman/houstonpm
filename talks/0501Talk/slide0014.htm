7 A Slightly More Complicated Class 

    
# The first derived Class

package ClassDerived;

use strict;
use warnings;

# old style: @ISA = [ "ClassSimple" ];
use base "ClassSimple";

my $objCount = 0;

sub  howMany
{
    $objCount;
}

sub  new
{
    my $self = shift;
    my $class = ref $self || $self;

    ++$objCount;
    
    bless ClassSimple::new(), $class;
}

sub  set
{
    my $self = shift;
    $self->{counter} = shift;
}

sub  incr
{
    my $self = shift;
    $self->add( 1 );
}

sub DESTROY
{
    my $self = shift;
    --$objCount;
}

1;
