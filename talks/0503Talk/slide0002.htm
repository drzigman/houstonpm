2 What are References?

    A reference is a scalar that refers or points to something.
    That something is retrieved through dereferencing.
    One way to generate a reference is with the backslash operator. 

    
# Reference to scalar

my $var = 1;

my $sref = \$var;
${$sref} = 5;

print "\$var = $var, \$\$sref = ${$sref}\n"; # prints $var = 5, $$sref = 5

# Reference to array

my @array = (1 .. 5);

my $aref = \@array;

${$aref}[1] = 10;
$aref->[3] = 30;

print "\@array = (@array), \@\$aref = (@{$aref})\n";
  # prints @array = (1 10 3 30 5), @$sref = (1 10 3 30 5)

# Reference to hash

my %hash = ( fred => "flintstone", barney => "rubble" );

my $href = \%hash;

${$href}{'larry'} = "wall";
foreach my $key (sort keys %{$href})
{
    print "$key = $href->{$key}\n";  # prints firstname = lastname for each key
}

# Reference to code

sub hello  { print "hello, world\n"; }

my $cref = \&hello;

&$cref();  # prints hello, world

$cref->(); # as does this.
