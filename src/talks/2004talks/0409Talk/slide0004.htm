4 Perl/Tk Hello World 

    
#!/usr/bin/perl

use Tk;
use strict;
use warnings;

my $mw = new MainWindow;
$mw->title( 'Hello' );

$mw->Label( -text => 'Hello World' )->pack;
$mw->Button( -text => "Done", -command => sub { exit } )->pack;

MainLoop;
