2 Traditional Methods

    system() 
    
   (0 == system( 'program arg1 arg2' ))
       or die "Unable to execute program: $?\n";
   # or
   (0 == system( 'program', 'arg1', 'arg2' ))
      or die "Unable to execute program: $?\n";
    exec() 
    
   exec( 'program arg1 arg2' )
      or die "Unable to execute program: $!\n";
    open() 
    
   open( my $fh, "program |" )
      or die "Unable to execute program: $!\n";
   # or
   open( my $fh, "| program" )
      or die "Unable to execute program: $!\n";
    backticks 
    
   my $output = qx/ls -l/;
   #or
   my @output = qx/ls -l/;
    IPC::Open2 
    
   open2( *README, *WRITEME, 'program' )
      or die "Unable to run program: $!\n";

   print WRITEME "Here's input to the program.\n";
   my $output = <README>;
   close( README );
   close( WRITEME );
