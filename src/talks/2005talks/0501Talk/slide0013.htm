6 Using the Class 

    
#!/usr/bin/perl

use Test::More tests => 6;

use strict;
use warnings;

use ClassSimple;

my $obj = ClassSimple::new();
isa_ok( $obj, 'ClassSimple' );

my $obj1 = new ClassSimple();
isa_ok( $obj1, 'ClassSimple' );

my $obj2 = $obj1->new();
isa_ok( $obj2, 'ClassSimple' );

is( $obj->get(), 0, "Initial value okay" );

$obj->add( 1 );
is( $obj->get(), 1, "One addition works." );

$obj->add( 1 );
is( $obj->get(), 2, "So does a second." );
