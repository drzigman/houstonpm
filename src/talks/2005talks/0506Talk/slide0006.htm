2.4 Still More Anchors

    \b - word boundary, (between a \w and a \W)
    \B - not a word boundary 
    
  my $string = "madam I am adam";
  $string =~ /adam/;   # matches at position 1
  $string =~ /\badam/; # matches last word
  $string =~ /m/;      # matches first letter
  $string =~ /\Bm/;    # matches end of first word
    \G - matches where last expression left off. 
    
  $string =~ /\w+\s+/;   # matches "madam "
  $string =~ /\G\w+\s+/; # matches "I "
