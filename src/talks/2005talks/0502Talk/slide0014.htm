12 A Complete Test::More example

Here's a simple test file for part of the List::Util module. 

    
#!/usr/bin/perl

use Test::More tests => 6;

use List::Util qw/max/;
use strict;
use warnings;

ok( !defined( max() ),   "empty list" );
is( max( 1,2,3,4,5 ), 5, "max item is last" );
is( max( 5,4,3,2,1 ), 5, "max item is first" );
is( max( 1,2,5,3,4 ), 5, "max item is middle" );
is( max( 5,5,5,5,5 ), 5, "all the same" );
is( max( 3 ),         3, "only one" );
